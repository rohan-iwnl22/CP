template<typename T>
int binPow(T a , T b){
    T ans = 1;
    while(b){
        if(b&1){
            ans = (ans * 1LL * a) % M;
        }
        a = (a*1LL*a)%M;
        b >>= 1;
    }
    
    return ans;
}
template <typename T>
T gcd(T a, T b) {
    while (b != 0) {
        T temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}
template <typename T>
T lcm(T a, T b) {
    return (a / gcd(a, b)) * b;  
}
